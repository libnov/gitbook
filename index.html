排序算法

1. 选择排序 O(n^2)

每次找最小的元素，和未排序元素交换位置

void selectSort(int arr[], int n){
  for(int i=0;i<n;i++){
    // 寻找 [i,n] 区间最小值
    int minIndex = i;
    for(int j=j+1；j<n;j++){
      if(arr[j]<arr[minIndex]){
        minIndex = j;
      }
    }
    //交换元素
    swap(arr[i],arr[minIndex]);
  }
}

2. 插入排序 O(n^2)

将一个元素插入到一个已经有序的数组里 （可以提前终止内层循环）

对于近乎有序的数据，性能更好，当已经是有序数组时会转换成O(n)

template<typename T>
void insertSort(T arr[], int n){
  for(int i=1;i<n;i++){
    // 寻找元素 arr[i] 插入的位置
    for(int j=i;j>0;j--){
      if(arr[j] < arr[j-1]){
        swap(arr[j],arr[j-1]);
      }
    }
  }
}

// 上一版本 每次比较后 做数据交换，交换的性能 要次于 数据赋值的性能
// 用赋值 代替 交换
void insertSort(T arr[], int n){
  for(int i=1;i<n;i++){
    // 寻找元素 arr[i] 插入的位置
    T e = arr[i];
    int j;
    for(int j=i;j>0;j--){
      if(arr[j-1] > e){
        arr[j] = arr[j-1];
      }
    }
    arr[j] = e;
  }
}

3. 归并排序 O(n*log n)

将数组不断二分，最终将有序的数组进行合并，需要耗费O(n)的空间

template<typename T>
void mergeSort(T arr[], int n){

  __mergeSort(arr,0,n-1);
}

//递归使用 归并排序 对 arr[l...r]范围进行排序
template<typename T>
__mergeSort(T arr[], int l,int r){
  if(l >= r){
    return;
  }

  int mid = (l+r)/2;
  __mergeSort(arr,l,mid);
  __mergeSort(arr,mid+1,r);

  __merge(arr,l,mid,r);
}
